/**
 * // This is the MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * class MountainArray {
 *   public:
 *     int get(int index);
 *     int length();
 * };
 */

class Solution {
public:
    int findInMountainArray(int target, MountainArray &mountainArr) {
        int n = mountainArr.length();
        int l = 0;
        int r = n-1;
        while (l < r){
            int mid = (l + r) / 2;
            if (mountainArr.get(mid) < mountainArr.get(mid+1)){
                l = mid + 1;
            }
            else {
                r = mid;
            }
        }
        int intermid = l;

        int left = 0;
        int right = intermid;

        while (left <= right){
            int mid = (left + right) / 2;
            int val = mountainArr.get(mid);
            if (val == target){
                return mid;
            }
            else if (val < target){
                left = mid + 1;   
            }
            else {
                right = mid - 1;
            }
        }

        left = intermid;
        right = n-1;
        while (left <= right) 
        {
            int mid = (left + right) / 2;
            int val = mountainArr.get(mid);
            if (val == target) {
                return mid;
            }
            else if (val > target) {
                left = mid + 1;
            }
            else {
                right = mid - 1;
            }
        }

        return -1;
    }
};
